
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() { return request.auth != null; }
    function isDeveloper(uid) {
      return exists(/databases/$(database)/documents/voters/$(uid)) &&
             get(/databases/$(database)/documents/voters/$(uid)).data.role == 'developer';
    }
    function isSelf(uid) { return request.auth.uid == uid; }

    match /candidates/{candidateId} {
      allow read: if isSignedIn();
      allow create, update, delete: if isDeveloper(request.auth.uid);
    }

    match /voters/{uid} {
      allow read, create, update, delete: if isDeveloper(request.auth.uid);
    }

    match /votes/{uid} {
      allow read, create, update: if isSelf(uid);
      allow delete: if isDeveloper(request.auth.uid);
    }

    match /results/{candidateId} {
      allow read, write: if isDeveloper(request.auth.uid);
    }
  }
}
